      /*SELECT   NVL (MAX (ID), 0) + 1
        INTO   v_sncode
        FROM   (SELECT   MAX (SNCODE) ID FROM MPUSNTAB
                UNION
                SELECT   MAX (CATALOGUE_ID) ID FROM CP_CATALOGUE);*/


      NEXTFREE.GETVALUE ('MAX_SNCODE', v_sncode);


      INSERT INTO mpusntab (sncode,
                            des,
                            shdes,
                            snind,
                            rec_version,
                            sntype,
                            pde_implicit_ind,
                            print_balance_ind,

                  p_print_balance_ind,
                  p_billing_ind,

              FROM   MPSSVTAB
             WHERE   SRVCODE = p_s1code AND SRVIND = 1 AND SCCODE = p_sccode;

         THEN
            v_s2code := 1;
         ELSE
            SELECT   SVCODE
              INTO   v_s2code
              FROM   MPSSVTAB

                               dirnum_npcode,
                               associate_dn,

                               associate_cug,

       WHERE   sncode = v_sncode;

      UPDATE   mpusntab
         SET   import_export_flag =
                  (import_export_flag + 2 - BITAND (import_export_flag, 2))
       WHERE   sncode = v_sncode;
   END;

   PROCEDURE INSERT_EVENT (p_des_event    VARCHAR2,
                           p_shservice    VARCHAR2,
                           p_evchg        VARCHAR2)
   IS
      v_evcode    NUMBER;
      v_sncode    NUMBER;
      v_rec_ver   NUMBER := 1;
   BEGIN
      SELECT   sncode
        INTO   v_sncode
        FROM   MPUSNTAB
       WHERE   shdes = p_shservice;

      SELECT   evcode
        INTO   v_evcode
        FROM   MPDEVTAB
       WHERE   des = p_des_event AND EVCHARGE = p_evchg;

      INSERT INTO MPULKEXN (SNCODE, EVCODE, REC_VERSION)
        VALUES   (v_sncode, v_evcode, v_rec_ver);

      UPDATE   MPUSNTAB
         SET   REC_VERSION = REC_VERSION + 1
       WHERE   SNCODE = v_sncode;

      UPDATE   MPUSNTAB
         SET   IMPORT_EXPORT_FLAG =
                  (IMPORT_EXPORT_FLAG + 2 - BITAND (IMPORT_EXPORT_FLAG, 2))
       WHERE   SNCODE = v_sncode;
   END;

                                 p_shdes        VARCHAR2,

        INTO   v_sncode

   PROCEDURE ADD_PARAM_VALUE (p_temp            VARCHAR2,
                              p_prm_no          NUMBER,
                              p_val_seq         NUMBER,
                              p_para_des        VARCHAR2,
                              p_prm_val_date    DATE,
                              p_prm_val_str     VARCHAR2,
                              p_prm_val_num     NUMBER,
                              p_prm_desc        VARCHAR2)
   IS
      v_temp_id   NUMBER;
      v_para_id   NUMBER;
   BEGIN
      SELECT   TEMPLATE_ID
        INTO   v_temp_id
        FROM   SERVICE_PRM_TEMPLATE_DEF
       WHERE   SHDES = p_temp;

      SELECT   PARAMETER_ID
        INTO   v_para_id
        FROM   MKT_PARAMETER
       WHERE   PRM_DES = p_para_des;

      INSERT INTO SERVICE_PRM_TEMPLATE (TEMPLATE_ID,
                                        SCCODE,
                                        PRM_NO,
                                        COMPLEX_SEQNO,
                                        COMPLEX_LEVEL,
                                        SIBLING_SEQNO,
                                        PARENT_SEQNO,
                                        PARAMETER_ID,
                                        VALUE_SEQNO,
                                        PRM_VALUE_DATE,
                                        PRM_VALUE_STRING,
                                        PRM_VALUE_NUMBER,
                                        PRM_DESCRIPTION)
        VALUES   (v_temp_id,
                  1,
                  p_prm_no,
                  1,
                  1,
                  1,
                  1,
                  v_para_id,
                  p_val_seq,
                  p_prm_val_date,
                  p_prm_val_str,
                  p_prm_val_num,
                  p_prm_desc);
   END;

   PROCEDURE NEW_CHG_QUAN (p_sn               VARCHAR2,
                           p_udm              VARCHAR2,
                           p_default          VARCHAR2,
                           p_char_quan_des    VARCHAR2,
                           p_unit_measure     VARCHAR2)
   IS
      v_sncode    NUMBER;
      v_udmcode   NUMBER;
   BEGIN
      SELECT   SNCODE
        INTO   v_sncode
        FROM   MPUSNTAB
       WHERE   SHDES = p_sn;

      IF (p_udm = 'MONEY')
      THEN
         SELECT   CHARGEABLE_QUANTITY_UDMCODE
           INTO   v_udmcode
           FROM   UDC_CHARGEABLE_QUANTITY_VIEW
          WHERE   LOGICAL_QUANTITY_CODE =
                     (SELECT   LOGICAL_QUANTITY_CODE
                        FROM   UDC_LOGICAL_QUANTITY
                       WHERE   LOGICAL_QUANTITY_SHDES = p_udm)
                  AND CHARGEABLE_QUANTITY_DES = p_char_quan_des;

         INSERT INTO SRV_CHARGEABLE_QUANTITY (
                                                 SNCODE,
                                                 CHARGEABLE_QUANTITY_UDMCODE,
                                                 CHARGEABLE_QUANTITY_DEF
                    )
           VALUES   (v_sncode, v_udmcode, p_default);
      ELSIF (p_udm = 'DUR')
      THEN
         SELECT   CHARGEABLE_QUANTITY_UDMCODE
           INTO   v_udmcode
           FROM   UDC_CHARGEABLE_QUANTITY_VIEW
          WHERE   LOGICAL_QUANTITY_SHDES = p_udm
                  AND UNIT_OF_MEASUREMENT_SHDES = p_unit_measure;

         INSERT INTO SRV_CHARGEABLE_QUANTITY (
                                                 SNCODE,
                                                 CHARGEABLE_QUANTITY_UDMCODE,
                                                 CHARGEABLE_QUANTITY_DEF
                    )
           VALUES   (v_sncode, v_udmcode, p_default);
      ELSIF (p_udm = 'DATAV')
      THEN
         SELECT   CHARGEABLE_QUANTITY_UDMCODE
           INTO   v_udmcode
           FROM   UDC_CHARGEABLE_QUANTITY_VIEW
          WHERE       LOGICAL_QUANTITY_SHDES = p_udm
                  AND CHARGEABLE_QUANTITY_DES = p_char_quan_des
                  AND UNIT_OF_MEASUREMENT_SHDES = p_unit_measure;

         INSERT INTO SRV_CHARGEABLE_QUANTITY (
                                                 SNCODE,
                                                 CHARGEABLE_QUANTITY_UDMCODE,
                                                 CHARGEABLE_QUANTITY_DEF
                    )
           VALUES   (v_sncode, v_udmcode, p_default);
      ELSE
         SELECT   CHARGEABLE_QUANTITY_UDMCODE
           INTO   v_udmcode
           FROM   UDC_CHARGEABLE_QUANTITY_VIEW
          WHERE   LOGICAL_QUANTITY_CODE =
                     (SELECT   LOGICAL_QUANTITY_CODE
                        FROM   UDC_LOGICAL_QUANTITY
                       WHERE   LOGICAL_QUANTITY_SHDES = p_udm);

         INSERT INTO SRV_CHARGEABLE_QUANTITY (
                                                 SNCODE,
                                                 CHARGEABLE_QUANTITY_UDMCODE,
                                                 CHARGEABLE_QUANTITY_DEF
                    )
           VALUES   (v_sncode, v_udmcode, p_default);
      END IF;
   END;

   PROCEDURE NEW_EXT_CODE (p_sn VARCHAR2, p_ext_code VARCHAR2)
   IS
      v_sncode   NUMBER;
   BEGIN
      SELECT   SNCODE
        INTO   v_sncode

        VALUES   (v_sncode, p_ext_code);
   END;

   PROCEDURE NEW_USG_TYPE (p_sn          VARCHAR2,
                           p_usg_type    VARCHAR2,
                           p_usg_def     VARCHAR2,
                           p_usg_mode    VARCHAR2,
                           p_ovw_sp      NUMBER)
   IS
      v_sncode   NUMBER;
      v_usg      NUMBER;
   BEGIN
      SELECT   SNCODE
        INTO   v_sncode
        FROM   MPUSNTAB
       WHERE   SHDES = p_sn;

      SELECT   USAGE_TYPE_ID
        INTO   v_usg
        FROM   UDC_USAGE_TYPE_TABLE
       WHERE   USAGE_TYPE_SHNAME = p_usg_type;

      INSERT INTO SRV_USAGE_TYPE (SNCODE,
                                  USAGE_TYPE_ID,
                                  USAGE_TYPE_DEF,
                                  USAGE_TYPE_MODE,
                                  OVW_SPCODE)
        VALUES   (v_sncode,
                  v_usg,
                  p_usg_def,
                  p_usg_mode,
                  p_ovw_sp);
   END;
